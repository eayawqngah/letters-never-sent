FROM rocker/r-ver:4.3.0

# Install system dependencies
RUN apt-get update && apt-get install -y \
    sqlite3 \
    libsqlite3-dev \
    libssl-dev \
    libcurl4-openssl-dev \
    libxml2-dev \
    && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Copy dependency file and install R packages
COPY init.R .
RUN Rscript init.R

# Copy all project files
COPY . .

# Expose port
EXPOSE 8080

# Run the application
CMD ["Rscript", "run.R"]
